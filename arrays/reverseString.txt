//User function Template for Java


class Reverse
{
    // Complete the function
    // str: input string
    public static String reverseWord(String str)
    {
        // Reverse the string str
        int n = str.length();
        int start = 0;
        int end = n-1;
        char[] ch = str.toCharArray(); 
        while(start < end){
            //swap first and end variable
            char temp = ch[start];
            ch[start] = ch[end];
            ch[end] = temp;
            end--;
            start++;
        }
        String s = new String(ch);
        return s;
    }
}
